# wechat4u.js

![](http://7xr8pm.com1.z0.glb.clouddn.com/nodeWechat.png)

wechat4u core版本，提供完善的API和持续同步功能，不处理和储存其他信息

## 运行

```
npm install
npm run core
```

## 启动实例

```javascript
'use strict'
import Wechat from './src/wechat.js'
import qrcode from 'qrcode-terminal'

let bot = new Wechat()

bot.on('uuid', uuid => {
  qrcode.generate('https://login.weixin.qq.com/l/' + uuid, {
    small: true
  })
  console.log('https://login.weixin.qq.com/qrcode/' + uuid)
})

bot.on('user-avatar', avatar => {
  console.log(avatar)
})

bot.on('login', () => {
  console.log('login')
})

bot.on('logout', () => {
  console.log('logout')
})

bot.on('contacts-updated', contacts => {
  console.log(Object.keys(contacts).length)
})

bot.on('text-message', msg => {
  console.log(`----------${msg.getDisplayTime()}----------`)
  console.log(bot.contacts[msg.FromUserName].getDisplayName() + ':\t' + msg.Content)
})

bot.on('emoticon-message', msg => {
  console.log(`----------${msg.getDisplayTime()}----------`)
  console.log(bot.contacts[msg.FromUserName].getDisplayName() + ':\t' + '表情信息，手机上查看')
})

bot.on('image-message', msg => {
  console.log(`----------${msg.getDisplayTime()}----------`)
  console.log(bot.contacts[msg.FromUserName].getDisplayName() + ':\t' + '图片信息，手机上查看')
})

bot.start()
```
